#build docker image
sudo docker build -f Dockerfile -t cmate .
# run docker container
sudo docker run -p 8080:8080 -t cmate .
# list docker images
docker image ls
# list all containers
docker ps -a
# remove all containers
docker rm $(docker ps -aq)
# remove dangling(without tag) images and containers
docker system prune
# ssh into docker container
docker exec -it <container_id> bash
